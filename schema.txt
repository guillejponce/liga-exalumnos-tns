CREATE TABLE public.competitions (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  season_id uuid NOT NULL,
  name text NOT NULL,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  CONSTRAINT competitions_pkey PRIMARY KEY (id),
  CONSTRAINT competitions_season_id_fkey FOREIGN KEY (season_id) REFERENCES public.seasons(id)
);
CREATE TABLE public.league_memberships (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  league_id uuid NOT NULL,
  user_id uuid NOT NULL,
  role USER-DEFINED NOT NULL DEFAULT 'viewer'::league_role,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  CONSTRAINT league_memberships_pkey PRIMARY KEY (id),
  CONSTRAINT league_memberships_league_id_fkey FOREIGN KEY (league_id) REFERENCES public.leagues(id)
);
CREATE TABLE public.leagues (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  name text NOT NULL,
  slug text NOT NULL UNIQUE,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  CONSTRAINT leagues_pkey PRIMARY KEY (id)
);
CREATE TABLE public.match_events (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  match_id uuid NOT NULL,
  team_season_id uuid NOT NULL,
  player_id uuid,
  type USER-DEFINED NOT NULL,
  minute integer,
  notes text,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  CONSTRAINT match_events_pkey PRIMARY KEY (id),
  CONSTRAINT match_events_match_id_fkey FOREIGN KEY (match_id) REFERENCES public.matches(id),
  CONSTRAINT match_events_team_season_id_fkey FOREIGN KEY (team_season_id) REFERENCES public.team_season(id),
  CONSTRAINT match_events_player_id_fkey FOREIGN KEY (player_id) REFERENCES public.players(id)
);
CREATE TABLE public.matches (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  stage_id uuid NOT NULL,
  group_id uuid,
  round integer,
  kickoff_at timestamp with time zone,
  home_team_season_id uuid NOT NULL,
  away_team_season_id uuid NOT NULL,
  home_score integer,
  away_score integer,
  status USER-DEFINED NOT NULL DEFAULT 'scheduled'::match_status,
  venue text,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  CONSTRAINT matches_pkey PRIMARY KEY (id),
  CONSTRAINT matches_stage_id_fkey FOREIGN KEY (stage_id) REFERENCES public.stages(id),
  CONSTRAINT matches_group_id_fkey FOREIGN KEY (group_id) REFERENCES public.stage_groups(id),
  CONSTRAINT matches_home_team_season_id_fkey FOREIGN KEY (home_team_season_id) REFERENCES public.team_season(id),
  CONSTRAINT matches_away_team_season_id_fkey FOREIGN KEY (away_team_season_id) REFERENCES public.team_season(id)
